 <script>
function addNumber() {
    const inputNumber = document.getElementById('inputNumber').value;
    const value1 = document.getElementById('value1');
    const value2 = document.getElementById('value2');

    const originalValue1 = parseFloat(value1.innerText.replace(/,/g, ''));
    const originalValue2 = parseFloat(value2.innerText.replace(/,/g, ''));

    if (inputNumber === '') {
        value1.innerText = originalValue1.toFixed(2).replace(/\B(?=(\d{3})+(?!\d))/g, ",");
        value2.innerText = originalValue2.toFixed(2).replace(/\B(?=(\d{3})+(?!\d))/g, ",");
    } else {
        const percentageValue1 = (originalValue1 * (1 + parseFloat(inputNumber) / 100)).toFixed(2).replace(/\B(?=(\d{3})+(?!\d))/g, ",");
        const percentageValue2 = (originalValue2 * (1 + parseFloat(inputNumber) / 100)).toFixed(2).replace(/\B(?=(\d{3})+(?!\d))/g, ",");
                value1.style.color = 'red';
                value2.style.color = 'red';
        value1.innerText = percentageValue1;
        value2.innerText = percentageValue2;
    }
}

        function refreshPage() {// 刷新页面
            location.reload();
        }

function toggleCalculateOrReset() {
    const button = document.getElementById('calculateOrResetButton');
    if (button.innerText === '计算') {
        addNumber();
        button.innerText = '重置';
    } else {
        refreshPage();
        button.innerText = '计算';
    }
}
